pipeline{
agent any
	stages{
		stage{
			steps('Build and Package'){
				echo 'Building and packaging the code using Maven'
				powershell 'mvn clean package'
				}
		
			post{
			success{
			echo 'Archiving the package.....'
			archiveArtifacts artifacts: '**/*.war', followSymlinks: false
			}

			}
			}
		}
}
